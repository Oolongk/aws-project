Frontend tải HTML/JavaScript từ S3 Static Website và gọi API Gateway để vẽ QMJ history bằng Chart.js.

| Bước | Hành động | Chi tiết kỹ thuật |
|---|---|---|
| 1 | Mở file `static/frontend/index.html` | Thay `YOUR_API_ID` bằng HTTP API Invoke URL |
| 2 | Amazon S3 → `stock-screener-web-tên-của-bạn` | Upload `index.html` |
| 3 | Bucket → Properties | Mở Static website hosting nếu dùng website endpoint |
| 4 | Index document | Nhập `index.html` |
| 5 | Permissions | Chỉ cấu hình public read cho web asset hoặc dùng CloudFront OAC |
| 6 | Mở website endpoint | Nhập ticker để test |

<script>
    // ==========================================
    // ⚠️ ĐIỀN LINK API GATEWAY CỦA BẠN VÀO ĐÂY
    // ==========================================
    const API_URL = "xxxxxxxxxxxxxxxxxxxxx";
    
    let allData = []; // Lưu toàn bộ dữ liệu lịch sử
    let dataTableInstance = null;

    function getScoreColorClass(score) {
        if (score === null || score === undefined) return "text-secondary";
        if (score >= 80) return "score-excellent";
        if (score >= 60) return "score-good";
        if (score >= 40) return "score-average";
        return "score-bad";
    }

    function formatNumber(num) {
        if (!num) return "-";
        return Number(num).toLocaleString('vi-VN', { maximumFractionDigits: 1 });
    }

    async function initApp() {
        try {
            const response = await fetch(API_URL);
            allData = await response.json();
            
            // Lọc data hợp lệ
            allData = allData.filter(s => s.ticker != null);
            
            // 1. Trích xuất danh sách các Ngày/Quý duy nhất (Xóa trùng lặp và Sắp xếp mới nhất lên đầu)
            const uniqueDates = [...new Set(allData.map(item => item.snapshot_date))].filter(Boolean).sort().reverse();
            
            // 2. Đổ các ngày vào Dropdown
            const dateSelector = document.getElementById("dateSelector");
            dateSelector.innerHTML = ""; // Xóa chữ "Đang tải..."
            
            if(uniqueDates.length > 0) {
                uniqueDates.forEach(date => {
                    let option = document.createElement("option");
                    option.value = date;
                    option.innerText = date;
                    dateSelector.appendChild(option);
                });

                // Lắng nghe sự kiện khi người dùng chọn một Quý khác
                dateSelector.addEventListener("change", function(e) {
                    renderTableForDate(e.target.value);
                });

                // Tự động hiển thị dữ liệu của ngày mới nhất
                renderTableForDate(uniqueDates[0]);
            } else {
                dateSelector.innerHTML = "<option>Không có dữ liệu</option>";
            }
            
            // Tắt màn hình loading
            document.getElementById("loading-screen").style.display = "none";

        } catch (error) {
            document.getElementById("loading-screen").innerHTML = `
                <h3 class="text-danger"><i class="fa-solid fa-triangle-exclamation"></i> Lỗi kết nối API</h3>
                <p class="text-secondary">${error.message}</p>
            `;
        }
    }

    // Hàm Lọc và Vẽ lại bảng dựa trên ngày được chọn
    function renderTableForDate(selectedDate) {
        // Lọc dữ liệu của ngày đang chọn
        let filteredData = allData.filter(item => item.snapshot_date === selectedDate);
        
        // Sắp xếp theo Rank
        filteredData.sort((a, b) => (a.qmj_rank || 999) - (b.qmj_rank || 999));

        // Hủy bảng cũ trước khi vẽ bảng mới (Ngăn lỗi DataTables)
        if (dataTableInstance !== null) {
            dataTableInstance.destroy();
        }

        const tbody = document.querySelector("#stockTable tbody");
        tbody.innerHTML = ""; // Xóa dữ liệu cũ
        
        filteredData.forEach(stock => {
            const tr = document.createElement("tr");
            
            let exchangeClass = "";
            if(stock.exchange === "HSX") exchangeClass = "exchange-HSX";
            else if(stock.exchange === "HNX") exchangeClass = "exchange-HNX";
            else exchangeClass = "exchange-UPCOM";

            tr.innerHTML = `
                <td class="text-center text-secondary fw-bold">${stock.qmj_rank || '-'}</td>
                <td><span class="ticker-badge ${exchangeClass}">${stock.ticker}</span></td>
                <td class="text-secondary small fw-bold">${stock.exchange || '-'}</td>
                <td class="text-truncate" style="max-width: 180px;" title="${stock.company_name || '-'}">${stock.company_name || '-'}</td>
                <td class="text-truncate" style="max-width: 150px;">${stock.industry_group || '-'}</td>
                <td class="text-truncate text-secondary" style="max-width: 150px;">${stock.sector_detail || '-'}</td>
                <td class="text-end fw-semibold">${formatNumber(stock.current_market_cap)}</td>
                <td class="text-end text-muted small">${formatNumber(stock.quarter_market_cap)}</td>
                
                <td class="text-center fs-6 border-start border-secondary ${getScoreColorClass(stock.qmj_score)}">${stock.qmj_score ? Number(stock.qmj_score).toFixed(1) : '-'}</td>
                <td class="text-center ${getScoreColorClass(stock.qmj_profitability)}">${stock.qmj_profitability ? Number(stock.qmj_profitability).toFixed(1) : '-'}</td>
                <td class="text-center ${getScoreColorClass(stock.qmj_growth)}">${stock.qmj_growth ? Number(stock.qmj_growth).toFixed(1) : '-'}</td>
                <td class="text-center ${getScoreColorClass(stock.qmj_safety)}">${stock.qmj_safety ? Number(stock.qmj_safety).toFixed(1) : '-'}</td>
                <td class="text-center ${getScoreColorClass(stock.value_recent_score)}">${stock.value_recent_score ? Number(stock.value_recent_score).toFixed(1) : '-'}</td>
                <td class="text-center ${getScoreColorClass(stock.momentum_recent_score)}">${stock.momentum_recent_score ? Number(stock.momentum_recent_score).toFixed(1) : '-'}</td>
            `;
            tbody.appendChild(tr);
        });

        // Khởi tạo lại DataTables
        dataTableInstance = $('#stockTable').DataTable({
            pageLength: 20,
            lengthMenu: [20, 50, 100, 500],
            scrollX: true, 
            language: {
                search: "🔍 Lọc cổ phiếu:",
                lengthMenu: "Hiển thị _MENU_ mã",
                info: "Đang hiển thị _START_ đến _END_ trên tổng _TOTAL_ mã",
                paginate: { next: "Sau", previous: "Trước" },
                zeroRecords: "Không tìm thấy mã cổ phiếu nào."
            },
            order: [[0, 'asc']], 
            columnDefs: [
                { orderable: false, targets: [3, 4, 5] } 
            ]
        });
    }

    window.onload = initApp;
</script>
</body>
</html>
