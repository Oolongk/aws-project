Backfill chạy local để nạp price history từ `2023-01-01` đến ngày hiện tại. Code lọc S&P 500 theo `MIN_MARKET_CAP = 10_000_000_000`, partition giá theo từng ngày giao dịch và ghi company info snapshot tại ngày chạy.

Backfill hiện tại không ghi `us_financials_raw.parquet`. Financials được daily ingestion nạp; transform code đã có fallback nếu financials chưa tồn tại.

````bash
# Tạo environment riêng cho local backfill.
python -m venv .venv
.\.venv\Scripts\Activate.ps1
````

## Chuẩn bị và chạy backfill

| Bước | Hành động | Chi tiết kỹ thuật |
|---|---|---|
| 1 | Tạo file `backfill_raw_ingestion.py` | Dùng source full bên dưới |
| 2 | Kiểm tra `BUCKET_NAME` | Phải là `stock-screener-raw-tên-của-bạn` |
| 3 | Đặt `MAX_TICKERS_FOR_TEST` | Test số nhỏ trước; dùng `None` khi chạy toàn bộ |
| 4 | Giữ credential ngoài source code | boto3 dùng AWS CLI profile/credential chain |
| 5 | Chạy Python script | Script gọi STS để verify identity trước khi crawl |
````bash
# Chạy local backfill.
python backfill_raw_ingestion.py
````

## Full source: `backfill_raw_ingestion.py`
````python
import io
import time
from datetime import datetime, timedelta

import boto3
import pandas as pd
import requests
import yfinance as yf
from botocore.exceptions import ClientError
from tqdm import tqdm

# =========================================================
# AWS S3 CONFIG
# =========================================================
BUCKET_NAME = "stock-screener-raw-tên-của-bạn"
RAW_ROOT = "raw_us_market"

# boto3 lấy credential từ AWS CLI profile, biến môi trường hoặc IAM Role.
# Không hard-code Access Key hoặc Secret Key trong source code.

# Phải đúng region bucket của bạn
AWS_REGION = "ap-southeast-1"

# =========================================================
# BACKFILL CONFIG
# =========================================================
MIN_MARKET_CAP = 10_000_000_000  # 10 tỷ USD

# Khoảng thời gian muốn backfill giá
BACKFILL_START_DATE = "2023-01-01"
BACKFILL_END_DATE = (datetime.now() + timedelta(days=1)).strftime("%Y-%m-%d")

# Test 30 mã trước. Đổi thành None để chạy toàn bộ S&P 500
MAX_TICKERS_FOR_TEST = None

# =========================================================
# AWS CLIENTS
# =========================================================
# Dùng default credential provider chain thay vì nhúng credential vào file.
session = boto3.Session(region_name=AWS_REGION)
s3 = session.client("s3")
sts = session.client("sts")


# =========================================================
# AWS CHECK
# =========================================================
def verify_aws_connection():
    try:
        identity = sts.get_caller_identity()

        print("✅ AWS authentication thành công")
        print(f"   Account: {identity['Account']}")
        print(f"   ARN: {identity['Arn']}")

    except ClientError as e:
        raise RuntimeError(
            f"❌ AWS credentials lỗi: {e.response['Error']['Code']}"
        ) from e


# =========================================================
# PARQUET + S3
# =========================================================
def to_parquet_bytes(df: pd.DataFrame) -> bytes:
    """
    Ghi Parquet tương thích AWS Glue:
    - Không để datetime64[ns] / TIMESTAMP(NANOS)
    - Các cột ngày được lưu dạng string YYYY-MM-DD
    """
    output_df = df.copy()

    if not isinstance(output_df.index, pd.RangeIndex):
        output_df = output_df.reset_index()

    # Đổi tên Date/Datetime từ yfinance thành price_date
    if "Date" in output_df.columns:
        output_df = output_df.rename(columns={"Date": "price_date"})

    if "Datetime" in output_df.columns:
        output_df = output_df.rename(columns={"Datetime": "price_date"})

    # Chuyển mọi datetime sang string để Glue không gặp TIMESTAMP(NANOS)
    for column_name in output_df.columns:
        if pd.api.types.is_datetime64_any_dtype(output_df[column_name]):
            date_series = output_df[column_name]

            # Bỏ timezone nhưng giữ nguyên ngày giao dịch
            if isinstance(date_series.dtype, pd.DatetimeTZDtype):
                date_series = date_series.dt.tz_localize(None)

            # price_date / snapshot_date chỉ cần ngày
            if column_name in ["price_date", "snapshot_date"]:
                output_df[column_name] = date_series.dt.strftime("%Y-%m-%d")
            else:
                output_df[column_name] = date_series.dt.strftime(
                    "%Y-%m-%d %H:%M:%S"
                )

    buffer = io.BytesIO()

    output_df.to_parquet(
        buffer,
        index=False,
        compression="snappy",
        engine="pyarrow",
        version="1.0"
    )

    return buffer.getvalue()


def put_df(df: pd.DataFrame, key: str):
    """Upload một DataFrame lên S3 dưới dạng Parquet."""
    if df.empty:
        print(f"⚠️ Bỏ qua {key}: DataFrame rỗng.")
        return

    try:
        s3.put_object(
            Bucket=BUCKET_NAME,
            Key=key,
            Body=to_parquet_bytes(df),
            ContentType="application/octet-stream"
        )

        print(f"✅ Uploaded: s3://{BUCKET_NAME}/{key} | {len(df)} dòng")

    except ClientError as e:
        raise RuntimeError(
            f"❌ Không thể upload S3.\n"
            f"Error: {e.response['Error']['Code']}\n"
            f"Key: {key}"
        ) from e


# =========================================================
# TICKER SOURCE
# =========================================================
def get_sp500_tickers():
    url = "https://en.wikipedia.org/wiki/List_of_S%26P_500_companies"

    headers = {
        "User-Agent": (
            "Mozilla/5.0 (Windows NT 10.0; Win64; x64) "
            "AppleWebKit/537.36 Chrome/120.0 Safari/537.36"
        )
    }

    response = requests.get(url, headers=headers, timeout=30)
    response.raise_for_status()

    df_sp500 = pd.read_html(response.text)[0]

    # yfinance dùng BRK-B thay vì BRK.B
    return [
        ticker.replace(".", "-")
        for ticker in df_sp500["Symbol"].tolist()
    ]

# =========================================================
# COMPANY INFO (ĐÃ FIX LỖI THIẾU CỘT SCHEMA)
# =========================================================
def get_company_info(ticker: str, ingestion_date: str):
    """
    Company info từ yfinance là dữ liệu hiện tại.
    Đã đồng bộ 100% schema với code Daily Ingestion để Spark không báo lỗi.
    """
    stock = yf.Ticker(ticker)

    try:
        fast_info = stock.fast_info or {}
        market_cap = fast_info.get("marketCap", 0)
    except Exception:
        fast_info = {}
        market_cap = 0

    if not market_cap or market_cap < MIN_MARKET_CAP:
        return None

    try:
        info = stock.info or {}
    except Exception:
        info = {}

    return {
        "ticker": ticker,
        "snapshot_date": ingestion_date,
        "company_name": info.get("shortName") or info.get("longName") or ticker,
        "exchange": info.get("exchange"),
        "sector": info.get("sector"),
        "industry": info.get("industry"),
        "country": info.get("country"),
        "currency": info.get("currency"),
        "website": info.get("website"),
        "current_market_cap": market_cap or info.get("marketCap"),
        "shares_outstanding": info.get("sharesOutstanding"),
        "enterprise_value": info.get("enterpriseValue"),
        "float_shares": info.get("floatShares"),
        "beta": info.get("beta"),
        "trailing_pe": info.get("trailingPE"),
        "return_on_equity": info.get("returnOnEquity"),
        "debt_to_equity": info.get("debtToEquity"),
        "earnings_growth": info.get("earningsGrowth"),
        "revenue_growth": info.get("revenueGrowth"),
        "ingested_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }


# =========================================================
# PRICE HISTORY (ĐÃ FIX LỖI SCHEMA)
# =========================================================
def get_price_history(ticker: str, ingestion_date: str):
    """
    Lấy giá lịch sử trong khoảng BACKFILL_START_DATE -> BACKFILL_END_DATE.
    Đồng bộ chuẩn Schema (tên cột viết thường) để Spark không báo lỗi.
    """
    stock = yf.Ticker(ticker)

    df_price = stock.history(
        start=BACKFILL_START_DATE,
        end=BACKFILL_END_DATE,
        auto_adjust=False
    )

    if df_price.empty:
        return pd.DataFrame()

    # Đưa index Date thành cột price_date
    df_price = df_price.reset_index()

    # Xử lý cột ngày tháng
    if "Date" in df_price.columns:
        df_price = df_price.rename(columns={"Date": "price_date"})
    if "Datetime" in df_price.columns:
        df_price = df_price.rename(columns={"Datetime": "price_date"})

    df_price["price_date"] = pd.to_datetime(df_price["price_date"]).dt.tz_localize(None)

    # ---------------------------------------------------------
    # FIX LỖI SCHEMA: Chuyển đổi tên cột yfinance -> AWS Glue
    # ---------------------------------------------------------
    rename_map = {
        "Open": "open",
        "High": "high",
        "Low": "low",
        "Close": "close",
        "Adj Close": "adj_close",
        "Volume": "volume",
        "Dividends": "dividends",
        "Stock Splits": "stock_splits"
    }
    
    # Rename các cột có sẵn
    df_price = df_price.rename(columns=rename_map)

    # Trường hợp thư viện yfinance phiên bản mới không trả về Adj Close khi auto_adjust=False
    if "adj_close" not in df_price.columns and "close" in df_price.columns:
        df_price["adj_close"] = df_price["close"]

    # Ép kiểu dữ liệu (Cast type) về float64 để tránh lỗi Integer vs Double trong Spark
    numeric_columns = ["open", "high", "low", "close", "adj_close", "volume", "dividends", "stock_splits"]
    for col in numeric_columns:
        if col in df_price.columns:
            df_price[col] = pd.to_numeric(df_price[col], errors="coerce").astype("float64")

    # Thêm siêu dữ liệu (Metadata)
    df_price["ticker"] = ticker
    df_price["snapshot_date"] = ingestion_date
    df_price["ingested_at"] = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    # Chỉ giữ lại các cột cần thiết theo đúng thứ tự
    expected_cols = [
        "price_date", "ticker", "open", "high", "low", "close", 
        "adj_close", "volume", "dividends", "stock_splits", 
        "snapshot_date", "ingested_at"
    ]
    
    # Lọc những cột thực sự tồn tại (tránh lỗi KeyError nếu yfinance thiếu cột)
    final_cols = [c for c in expected_cols if c in df_price.columns]
    df_price = df_price[final_cols]

    return df_price


# =========================================================
# BACKFILL PARTITION UPLOAD
# =========================================================
def upload_prices_partitioned_by_date(df_prices: pd.DataFrame):
    """
    Upload theo từng ngày giao dịch.

    Ví dụ:
    raw_us_market/2023/01/03/us_prices_raw.parquet
    raw_us_market/2023/01/04/us_prices_raw.parquet
    """
    if df_prices.empty:
        print("⚠️ Không có dữ liệu giá để partition.")
        return

    df_prices = df_prices.copy()

    df_prices["price_date"] = pd.to_datetime(
        df_prices["price_date"],
        errors="coerce"
    )

    df_prices = df_prices.dropna(subset=["price_date"])

    grouped = df_prices.groupby(
        df_prices["price_date"].dt.normalize(),
        sort=True
    )

    print(f"\n☁️ Đang upload {len(grouped)} partition ngày...")

    for trade_date, df_one_day in tqdm(
        grouped,
        total=len(grouped),
        desc="Upload partition ngày"
    ):
        yyyy = trade_date.strftime("%Y")
        mm = trade_date.strftime("%m")
        dd = trade_date.strftime("%d")

        key = (
            f"{RAW_ROOT}/"
            f"{yyyy}/{mm}/{dd}/"
            f"us_prices_raw.parquet"
        )

        put_df(df_one_day, key)


def upload_company_info_current_snapshot(
    df_company_info: pd.DataFrame,
    ingestion_datetime: datetime
):
    """
    Company info chỉ là snapshot hiện tại,
    nên lưu theo ngày pipeline chạy.
    ĐÃ BỔ SUNG ÉP KIỂU FLOAT64 ĐỂ SPARK KHÔNG BÁO LỖI INT64 vs DOUBLE
    """
    if df_company_info.empty:
        return

    output_df = df_company_info.copy()

    # Bắt buộc ép toàn bộ các cột metric thành float64 (Double trên Spark)
    numeric_columns = [
        "current_market_cap", "shares_outstanding", "enterprise_value",
        "float_shares", "beta", "trailing_pe", "return_on_equity",
        "debt_to_equity", "earnings_growth", "revenue_growth"
    ]

    for col in numeric_columns:
        if col in output_df.columns:
            output_df[col] = pd.to_numeric(output_df[col], errors="coerce").astype("float64")

    yyyy = ingestion_datetime.strftime("%Y")
    mm = ingestion_datetime.strftime("%m")
    dd = ingestion_datetime.strftime("%d")

    key = (
        f"{RAW_ROOT}/"
        f"{yyyy}/{mm}/{dd}/"
        f"us_company_info.parquet"
    )

    put_df(output_df, key)


# =========================================================
# MAIN PIPELINE
# =========================================================
def main():
    print("=" * 70)
    print("🚀 STOCK SCREENER - BACKFILL RAW INGESTION")
    print("=" * 70)

    verify_aws_connection()

    run_datetime = datetime.now()
    ingestion_date = run_datetime.strftime("%Y-%m-%d")

    print(f"\n📅 Backfill start: {BACKFILL_START_DATE}")
    print(f"📅 Backfill end:   {BACKFILL_END_DATE}")

    # -----------------------------------------------------
    # BƯỚC 1: LẤY TICKER
    # -----------------------------------------------------
    print("\n🚀 BƯỚC 1: LẤY DANH SÁCH S&P 500")

    all_tickers = get_sp500_tickers()

    if MAX_TICKERS_FOR_TEST is not None:
        all_tickers = all_tickers[:MAX_TICKERS_FOR_TEST]

    print(f"✅ Số ticker sẽ xử lý: {len(all_tickers)}")

    # -----------------------------------------------------
    # BƯỚC 2: FILTER THEO MARKET CAP
    # -----------------------------------------------------
    print("\n🚀 BƯỚC 2: LỌC TICKER THEO MARKET CAP")

    valid_tickers = []
    all_company_info = []

    for ticker in tqdm(all_tickers, desc="Đang lọc ticker"):
        try:
            company_info = get_company_info(
                ticker=ticker,
                ingestion_date=ingestion_date
            )

            if company_info is not None:
                valid_tickers.append(ticker)
                all_company_info.append(company_info)

        except Exception as e:
            print(f"\n⚠️ Lỗi company info {ticker}: {e}")

        time.sleep(0.2)

    print(f"\n🎯 Có {len(valid_tickers)} ticker đạt điều kiện.")

    # -----------------------------------------------------
    # BƯỚC 3: BACKFILL PRICE HISTORY
    # -----------------------------------------------------
    print("\n🚀 BƯỚC 3: BACKFILL LỊCH SỬ GIÁ")

    all_prices = []

    for ticker in tqdm(valid_tickers, desc="Đang lấy lịch sử giá"):
        try:
            df_price = get_price_history(
                ticker=ticker,
                ingestion_date=ingestion_date
            )

            if not df_price.empty:
                all_prices.append(df_price)

        except Exception as e:
            print(f"\n⚠️ Lỗi lấy giá {ticker}: {e}")

        time.sleep(0.2)

    # -----------------------------------------------------
    # BƯỚC 4: UPLOAD S3
    # -----------------------------------------------------
    print("\n☁️ BƯỚC 4: UPLOAD LÊN S3")

    if all_prices:
        df_prices = pd.concat(all_prices, ignore_index=True)

        print(f"✅ Tổng số dòng price: {len(df_prices):,}")

        upload_prices_partitioned_by_date(df_prices)
    else:
        print("⚠️ Không có dữ liệu price để upload.")

    if all_company_info:
        df_company_info = pd.DataFrame(all_company_info)

        upload_company_info_current_snapshot(
            df_company_info=df_company_info,
            ingestion_datetime=run_datetime
        )
    else:
        print("⚠️ Không có company info để upload.")

    print("\n🎉 HOÀN TẤT BACKFILL!")
    print(f"📁 Bucket: s3://{BUCKET_NAME}/{RAW_ROOT}/")


if __name__ == "__main__":
    main()
````
> **Tư duy kiến trúc sư – idempotency.** Backfill ghi cùng object key cho cùng `price_date`, vì vậy rerun sẽ overwrite data ở partition tương ứng thay vì tạo rác. Rerun full chỉ nên thực hiện khi cần sửa dữ liệu lịch sử hoặc schema.
