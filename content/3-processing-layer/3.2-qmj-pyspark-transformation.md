Source transform hiện tại chỉ đọc S3 Raw và ghi S3 Processed. Không có lệnh ghi DynamoDB trực tiếp; Chương 4 dùng Publisher Job riêng để đọc Processed Parquet và publish sang Serving Layer.

## Bảy stage trong log source

| Stage | Log | Logic thực tế |
|---|---|---|
| 1 | `Processing Company Info` | Chọn snapshot info gần nhất hợp lệ tại/before `AS_OF_DATE` |
| 2 | `Processing Prices` | Dedupe price, daily return, annualized volatility, 12-month momentum |
| 3 | `Processing Financials` | Dedupe reports, annualized ROE, debt-to-equity và growth; có fallback |
| 4 | `Merging Data` | Join info, price metrics và financial metrics theo ticker |
| 5 | `Calculating Percentiles` | Percentile 0–100, safety score và QMJ score |
| 6 | `Building Final Table` | Rank giảm dần theo `qmj_score` |
| 7 | `Writing Processed Parquet` | Snappy Parquet theo `as_of_date` |

## Công thức trong source
````text
profitability_raw = financial_roe_raw
                    hoặc return_on_equity từ company info

growth_raw = financial_growth_raw
             hoặc earnings_growth
             hoặc revenue_growth

qmj_safety = average(safety_volatility_score, safety_debt_score)
qmj_score  = average(qmj_profitability, qmj_growth, qmj_safety)
````
`value_recent_score` và `momentum_recent_score` được tính/xuất ra, nhưng không tham gia công thức `qmj_score` hiện tại.

## Hành vi cần biết

| Hành vi source | Ý nghĩa |
|---|---|
| `recursiveFileLookup=true` | Quét mọi partition Raw Zone |
| `pathGlobFilter` | Đọc đúng từng loại file Parquet |
| Explicit `StructType` | Tránh Spark merge schema giữa backfill và daily files |
| `mergeSchema=false` | Không merge schema tự động |
| Financial fallback | Job có thể tiếp tục nếu file financials chưa tồn tại/đọc lỗi |
| Missing score = `50.0` | Neutral score cho metric thiếu sau percentile calculation |
| `repartition(1)` | Chỉ một part output; hợp lý dataset nhỏ nhưng không scale lớn |
| `MIN_PRICE_OBSERVATIONS` | Có argument nhưng current source chưa dùng để filter |

## Full source: `qmj_transform.py`
````python
import argparse
from datetime import datetime, timedelta

from awsglue.context import GlueContext
from pyspark.context import SparkContext
from pyspark.sql import Window
from pyspark.sql.functions import (
    abs,
    coalesce,
    col,
    count,
    lag,
    lit,
    percent_rank,
    row_number,
    round as spark_round,
    sqrt,
    stddev_samp,
    to_date,
    when
)
from pyspark.sql.types import (
    StructType,
    StructField,
    StringType,
    DoubleType
)

DEFAULT_RAW_BUCKET = "stock-screener-raw-tên-của-bạn"
DEFAULT_RAW_PREFIX = "raw_us_market"
DEFAULT_PROCESSED_BUCKET = "stock-screener-processed-tên-của-bạn"
DEFAULT_PROCESSED_PREFIX = "processed_qmj"


def get_args():
    parser = argparse.ArgumentParser()

    parser.add_argument("--RAW_BUCKET", default=DEFAULT_RAW_BUCKET)
    parser.add_argument("--RAW_PREFIX", default=DEFAULT_RAW_PREFIX)
    parser.add_argument("--PROCESSED_BUCKET", default=DEFAULT_PROCESSED_BUCKET)
    parser.add_argument("--PROCESSED_PREFIX", default=DEFAULT_PROCESSED_PREFIX)
    parser.add_argument(
        "--AS_OF_DATE",
        default=datetime.utcnow().strftime("%Y-%m-%d")
    )
    parser.add_argument("--LOOKBACK_DAYS", type=int, default=800)
    parser.add_argument("--MIN_PRICE_OBSERVATIONS", type=int, default=200)

    args, _ = parser.parse_known_args()
    return args


PRICE_SCHEMA = StructType([
    StructField("price_date", StringType(), True),
    StructField("ticker", StringType(), True),
    StructField("open", DoubleType(), True),
    StructField("high", DoubleType(), True),
    StructField("low", DoubleType(), True),
    StructField("close", DoubleType(), True),
    StructField("adj_close", DoubleType(), True),
    StructField("volume", DoubleType(), True),
    StructField("dividends", DoubleType(), True),
    StructField("stock_splits", DoubleType(), True),
    StructField("snapshot_date", StringType(), True),
    StructField("ingested_at", StringType(), True)
])

INFO_SCHEMA = StructType([
    StructField("ticker", StringType(), True),
    StructField("snapshot_date", StringType(), True),
    StructField("company_name", StringType(), True),
    StructField("exchange", StringType(), True),
    StructField("sector", StringType(), True),
    StructField("industry", StringType(), True),
    StructField("country", StringType(), True),
    StructField("currency", StringType(), True),
    StructField("website", StringType(), True),
    StructField("current_market_cap", DoubleType(), True),
    StructField("shares_outstanding", DoubleType(), True),
    StructField("enterprise_value", DoubleType(), True),
    StructField("float_shares", DoubleType(), True),
    StructField("beta", DoubleType(), True),
    StructField("trailing_pe", DoubleType(), True),
    StructField("return_on_equity", DoubleType(), True),
    StructField("debt_to_equity", DoubleType(), True),
    StructField("earnings_growth", DoubleType(), True),
    StructField("revenue_growth", DoubleType(), True),
    StructField("ingested_at", StringType(), True)
])

FINANCIAL_SCHEMA = StructType([
    StructField("ticker", StringType(), True),
    StructField("report_date", StringType(), True),
    StructField("snapshot_date", StringType(), True),
    StructField("period_type", StringType(), True),
    StructField("revenue", DoubleType(), True),
    StructField("net_income", DoubleType(), True),
    StructField("shareholders_equity", DoubleType(), True),
    StructField("total_assets", DoubleType(), True),
    StructField("total_debt", DoubleType(), True),
    StructField("operating_cash_flow", DoubleType(), True),
    StructField("capital_expenditure", DoubleType(), True),
    StructField("free_cash_flow", DoubleType(), True),
    StructField("stock_repurchased", DoubleType(), True),
    StructField("stock_issued", DoubleType(), True),
    StructField("ingested_at", StringType(), True)
])


def get_schema_for_file(file_name):
    if "us_prices_raw" in file_name:
        return PRICE_SCHEMA

    if "us_company_info" in file_name:
        return INFO_SCHEMA

    if "us_financials_raw" in file_name:
        return FINANCIAL_SCHEMA

    return None


def read_required_parquet(spark, root_path, file_name):
    schema = get_schema_for_file(file_name)

    return (
        spark.read
        .schema(schema)
        .option("recursiveFileLookup", "true")
        .option("pathGlobFilter", file_name)
        .parquet(root_path)
    )


def add_percentile_score(
    df,
    metric_column,
    score_column,
    higher_is_better
):
    valid_metrics = (
        df
        .select(
            col("ticker"),
            col(metric_column).cast("double").alias("_metric")
        )
        .filter(col("_metric").isNotNull())
    )

    metric_window = Window.orderBy(
        col("_metric").asc()
        if higher_is_better
        else col("_metric").desc()
    )

    scores = (
        valid_metrics
        .withColumn(
            score_column,
            spark_round(
                percent_rank().over(metric_window) * lit(100.0),
                2
            )
        )
        .select("ticker", score_column)
    )

    return df.join(scores, "ticker", "left")


def main():
    args = get_args()

    as_of_datetime = datetime.strptime(
        args.AS_OF_DATE,
        "%Y-%m-%d"
    )

    lookback_start = (
        as_of_datetime - timedelta(days=args.LOOKBACK_DAYS)
    ).strftime("%Y-%m-%d")

    raw_root = (
        f"s3://{args.RAW_BUCKET}/"
        f"{args.RAW_PREFIX}/"
    )

    output_path = (
        f"s3://{args.PROCESSED_BUCKET}/"
        f"{args.PROCESSED_PREFIX}/"
        f"as_of_date={args.AS_OF_DATE}/"
    )

    sc = SparkContext.getOrCreate()
    glue_context = GlueContext(sc)
    spark = glue_context.spark_session

    spark.conf.set("spark.sql.parquet.mergeSchema", "false")
    spark.conf.set("spark.sql.parquet.enableVectorizedReader", "false")

    print("=" * 80)
    print("QMJ TRANSFORM JOB")
    print(f"AS_OF_DATE: {args.AS_OF_DATE}")
    print("=" * 80)

    # Bước 1: chọn snapshot company info hợp lệ nhất cho AS_OF_DATE.
    print("[1/7] Processing Company Info...")

    df_info_raw = read_required_parquet(
        spark,
        raw_root,
        "us_company_info.parquet"
    )

    df_info = (
        df_info_raw
        .select(
            col("ticker").cast("string"),
            to_date(col("snapshot_date")).alias("snapshot_date"),
            col("company_name").cast("string"),
            col("exchange").cast("string"),
            col("sector").alias("industry_group"),
            col("industry").alias("sector_detail"),
            col("current_market_cap").cast("double"),
            col("shares_outstanding").cast("double"),
            col("trailing_pe").cast("double"),
            col("return_on_equity").cast("double"),
            col("debt_to_equity").cast("double"),
            col("earnings_growth").cast("double"),
            col("revenue_growth").cast("double"),
            col("ingested_at")
        )
        .filter(
            col("ticker").isNotNull()
            & col("snapshot_date").isNotNull()
        )
        .withColumn(
            "_has_valid_historical_snapshot",
            when(
                col("snapshot_date")
                <= lit(args.AS_OF_DATE).cast("date"),
                lit(1)
            ).otherwise(lit(0))
        )
    )

    info_window = Window.partitionBy("ticker").orderBy(
        col("_has_valid_historical_snapshot").desc(),
        col("snapshot_date").desc(),
        col("ingested_at").desc()
    )

    df_info = (
        df_info
        .withColumn("_rank", row_number().over(info_window))
        .filter(col("_rank") == 1)
        .drop("_rank", "_has_valid_historical_snapshot")
    )

    df_info.cache()

    # Bước 2: dedupe prices, tính volatility và momentum 12 tháng.
    print("[2/7] Processing Prices...")

    df_prices_raw = read_required_parquet(
        spark,
        raw_root,
        "us_prices_raw.parquet"
    )

    df_prices = (
        df_prices_raw
        .select(
            col("ticker").cast("string"),
            to_date(col("price_date")).alias("price_date"),
            col("adj_close").cast("double"),
            col("close").cast("double"),
            col("volume").cast("double"),
            col("ingested_at")
        )
        .filter(
            col("ticker").isNotNull()
            & col("price_date").isNotNull()
            & col("adj_close").isNotNull()
        )
        .filter(
            (col("price_date") >= lit(lookback_start).cast("date"))
            & (
                col("price_date")
                <= lit(args.AS_OF_DATE).cast("date")
            )
        )
    )

    price_dedupe_window = Window.partitionBy(
        "ticker",
        "price_date"
    ).orderBy(
        col("ingested_at").desc()
    )

    df_prices = (
        df_prices
        .withColumn("_rank", row_number().over(price_dedupe_window))
        .filter(col("_rank") == 1)
        .drop("_rank")
    )

    price_window = Window.partitionBy("ticker").orderBy(
        col("price_date").asc()
    )

    df_returns = (
        df_prices
        .withColumn(
            "prev_close",
            lag("adj_close", 1).over(price_window)
        )
        .withColumn(
            "close_252",
            lag("adj_close", 252).over(price_window)
        )
        .withColumn(
            "daily_return",
            when(
                col("prev_close") > 0,
                (col("adj_close") / col("prev_close")) - lit(1.0)
            )
        )
    )

    df_price_coverage = (
        df_returns
        .groupBy("ticker")
        .agg(
            count("*").alias("price_observations")
        )
    )

    df_vol = (
        df_returns
        .filter(col("daily_return").isNotNull())
        .groupBy("ticker")
        .agg(
            (
                stddev_samp("daily_return")
                * sqrt(lit(252.0))
            ).alias("annualized_volatility")
        )
    )

    latest_price_window = Window.partitionBy("ticker").orderBy(
        col("price_date").desc()
    )

    df_latest = (
        df_returns
        .withColumn(
            "_latest_rank",
            row_number().over(latest_price_window)
        )
        .filter(col("_latest_rank") == 1)
        .select(
            "ticker",
            col("price_date").alias("latest_price_date"),
            col("adj_close").alias("latest_adj_close"),
            "close_252"
        )
        .withColumn(
            "momentum_12m_raw",
            when(
                col("close_252") > 0,
                (
                    col("latest_adj_close")
                    / col("close_252")
                ) - lit(1.0)
            )
        )
    )

    df_price_metrics = (
        df_latest
        .join(df_vol, "ticker", "left")
        .join(df_price_coverage, "ticker", "left")
    )

    df_price_metrics.cache()

    # Bước 3: financials optional; job fallback nếu không có file hoặc API snapshot.
    print("[3/7] Processing Financials...")

    try:
        raw_fin = read_required_parquet(
            spark,
            raw_root,
            "us_financials_raw.parquet"
        )

        df_fin = (
            raw_fin
            .select(
                col("ticker").cast("string"),
                to_date(col("report_date")).alias("report_date"),
                to_date(col("snapshot_date")).alias("snapshot_date"),
                col("period_type").cast("string"),
                col("revenue").cast("double"),
                col("net_income").cast("double"),
                col("shareholders_equity").cast("double"),
                col("total_debt").cast("double"),
                col("ingested_at")
            )
            .filter(
                col("ticker").isNotNull()
                & col("report_date").isNotNull()
            )
            .filter(
                col("report_date")
                <= lit(args.AS_OF_DATE).cast("date")
            )
        )

        fin_dedupe_window = Window.partitionBy(
            "ticker",
            "report_date",
            "period_type"
        ).orderBy(
            col("snapshot_date").desc(),
            col("ingested_at").desc()
        )

        df_fin = (
            df_fin
            .withColumn(
                "_rank",
                row_number().over(fin_dedupe_window)
            )
            .filter(col("_rank") == 1)
            .drop("_rank")
        )

        period_window = Window.partitionBy(
            "ticker",
            "report_date"
        ).orderBy(
            col("period_type").desc()
        )

        df_fin = (
            df_fin
            .withColumn(
                "_period_rank",
                row_number().over(period_window)
            )
            .filter(col("_period_rank") == 1)
            .drop("_period_rank")
        )

        fin_window = Window.partitionBy("ticker").orderBy(
            col("report_date").asc()
        )

        df_fin = (
            df_fin
            .withColumn(
                "rev_4",
                lag("revenue", 4).over(fin_window)
            )
            .withColumn(
                "financial_roe_raw",
                when(
                    col("shareholders_equity") > 0,
                    (
                        col("net_income") * lit(4.0)
                    ) / col("shareholders_equity")
                )
            )
            .withColumn(
                "financial_debt_to_equity",
                when(
                    col("shareholders_equity") > 0,
                    col("total_debt")
                    / col("shareholders_equity")
                )
            )
            .withColumn(
                "financial_growth_raw",
                when(
                    col("rev_4") > 0,
                    (
                        col("revenue")
                        / col("rev_4")
                    ) - lit(1.0)
                )
            )
        )

        latest_fin_window = Window.partitionBy("ticker").orderBy(
            col("report_date").desc()
        )

        df_financial_metrics = (
            df_fin
            .withColumn(
                "_latest_rank",
                row_number().over(latest_fin_window)
            )
            .filter(col("_latest_rank") == 1)
            .select(
                "ticker",
                "financial_roe_raw",
                "financial_debt_to_equity",
                "financial_growth_raw"
            )
        )

        df_financial_metrics.cache()

    except Exception as error:
        print(f"Không đọc được Financials: {error}")
        df_financial_metrics = None

    # Bước 4: join mọi metric theo grain ticker.
    print("[4/7] Merging Data...")

    df_master = df_info.join(
        df_price_metrics,
        "ticker",
        "left"
    )

    if df_financial_metrics is not None:
        df_master = df_master.join(
            df_financial_metrics,
            "ticker",
            "left"
        )
    else:
        df_master = (
            df_master
            .withColumn(
                "financial_roe_raw",
                lit(None).cast("double")
            )
            .withColumn(
                "financial_debt_to_equity",
                lit(None).cast("double")
            )
            .withColumn(
                "financial_growth_raw",
                lit(None).cast("double")
            )
        )

    df_master = (
        df_master
        .withColumn(
            "profitability_raw",
            coalesce(
                col("financial_roe_raw"),
                when(
                    abs(col("return_on_equity")) > 2.0,
                    col("return_on_equity") / lit(100.0)
                ).otherwise(col("return_on_equity"))
            )
        )
        .withColumn(
            "growth_raw",
            coalesce(
                col("financial_growth_raw"),
                col("earnings_growth"),
                col("revenue_growth")
            )
        )
        .withColumn(
            "debt_to_equity_raw",
            coalesce(
                col("financial_debt_to_equity"),
                when(
                    abs(col("debt_to_equity")) > 10.0,
                    col("debt_to_equity") / lit(100.0)
                ).otherwise(col("debt_to_equity"))
            )
        )
        .withColumn(
            "positive_trailing_pe",
            when(
                col("trailing_pe") > 0,
                col("trailing_pe")
            )
        )
        .withColumn(
            "quarter_market_cap",
            when(
                (col("shares_outstanding") > 0)
                & (col("latest_adj_close") > 0),
                col("shares_outstanding")
                * col("latest_adj_close")
            ).otherwise(col("current_market_cap"))
        )
    )

    # Bước 5: chuyển metric raw thành percentile score 0-100.
    print("[5/7] Calculating Percentiles...")

    df_master = add_percentile_score(
        df_master,
        "profitability_raw",
        "qmj_profitability",
        True
    )

    df_master = add_percentile_score(
        df_master,
        "growth_raw",
        "qmj_growth",
        True
    )

    df_master = add_percentile_score(
        df_master,
        "annualized_volatility",
        "safety_volatility_score",
        False
    )

    df_master = add_percentile_score(
        df_master,
        "debt_to_equity_raw",
        "safety_debt_score",
        False
    )

    df_master = add_percentile_score(
        df_master,
        "positive_trailing_pe",
        "value_recent_score",
        False
    )

    df_master = add_percentile_score(
        df_master,
        "momentum_12m_raw",
        "momentum_recent_score",
        True
    )

    df_master = df_master.withColumn(
        "qmj_safety",
        when(
            col("safety_volatility_score").isNotNull()
            & col("safety_debt_score").isNotNull(),
            spark_round(
                (
                    col("safety_volatility_score")
                    + col("safety_debt_score")
                ) / lit(2.0),
                2
            )
        ).otherwise(
            coalesce(
                col("safety_volatility_score"),
                col("safety_debt_score")
            )
        )
    )

    for score_column in [
        "qmj_profitability",
        "qmj_growth",
        "qmj_safety",
        "value_recent_score",
        "momentum_recent_score"
    ]:
        df_master = df_master.withColumn(
            score_column,
            coalesce(col(score_column), lit(50.0))
        )

    df_master = df_master.withColumn(
        "qmj_score",
        spark_round(
            (
                col("qmj_profitability")
                + col("qmj_growth")
                + col("qmj_safety")
            ) / lit(3.0),
            2
        )
    )

    # Bước 6: rank qmj_score và chọn đúng output contract.
    print("[6/7] Building Final Table...")

    qmj_rank_window = Window.orderBy(
        col("qmj_score").desc(),
        col("ticker").asc()
    )

    df_final = (
        df_master
        .withColumn(
            "qmj_rank",
            row_number().over(qmj_rank_window).cast("int")
        )
        .select(
            "ticker",
            coalesce(
                col("company_name"),
                col("ticker")
            ).alias("company_name"),
            "qmj_rank",
            "exchange",
            "industry_group",
            "sector_detail",
            "qmj_score",
            "qmj_profitability",
            "qmj_growth",
            "qmj_safety",
            "current_market_cap",
            spark_round(
                col("quarter_market_cap"),
                2
            ).alias("quarter_market_cap"),
            "value_recent_score",
            "momentum_recent_score"
        )
    )

    # Bước 7: ghi Parquet Snappy theo partition as_of_date.
    print("[7/7] Writing Processed Parquet...")

    (
        df_final
        .repartition(1)
        .orderBy("qmj_rank")
        .write
        .mode("overwrite")
        .option("compression", "snappy")
        .parquet(output_path)
    )

    print("=" * 80)
    print("QMJ TRANSFORM HOÀN TẤT")
    print("=" * 80)


if __name__ == "__main__":
    main()
````
